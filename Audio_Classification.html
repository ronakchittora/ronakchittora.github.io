def graph_spectrogram(wav_file):
    rate, data = wavfile.read(wav_file)
    nfft = 256 # Length of each window segment
    fs = 8000 # Sampling frequencies
    noverlap = 120 # Overlap between windows
    nchannels = data.ndim
    if nchannels == 1:
        pxx, freqs, bins, im = plt.specgram(data, nfft, fs, noverlap = noverlap)
    elif nchannels == 2:
        pxx, freqs, bins, im = plt.specgram(data[:,0], nfft, fs, noverlap = noverlap)
    return pxx

def get_spectrogram(wav_file):
    rate, data = wavfile.read(wav_file)
    nfft = 256 # Length of each window segment
    fs = 8000 # Sampling frequencies
    noverlap = 120 # Overlap between windows
    nchannels = data.ndim
    if nchannels == 1:
        freqs, bins, Sxx = spectrogram(data, fs=fs, nfft=nfft, noverlap = noverlap)
    elif nchannels == 2:
        freqs, bins, Sxx = spectrogram(data[:,0], fs=fs, nfft=nfft, noverlap = noverlap)
    return Sxx

# function to get 1 sec of random backgroud noise
# may need to add a silent segment to some to keep dimensions same -- AudioSegment.silent(duration=1000 or len(wav_file))
random.seed(42)

def add_noise(audio_file):
    orig_audio = AudioSegment.from_file(audio_file)
    noise_type = random.randint(0,background_noise.shape[1]-1)
    back_noise = AudioSegment.from_file(background_noise['path'].iloc[noise_type])
    if len(back_noise) > len(orig_audio):
        start_of_noise = random.randrange(len(back_noise)-len(orig_audio))
        audio = orig_audio.overlay(back_noise[start_of_noise:start_of_noise+len(orig_audio)])
        return audio
    else:
        audio = orig_audio.overlay(back_noise, loop=true)
        return audio

def get_training_data():
    k = int(0.3*train.shape[0])
    indexes_for_adding_noise = random.sample(range(train.shape[0]), k=k)
    
    for ind in indexes_for_adding_noise:
        audio = add_noise(train['path'].iloc[ind])
        audio.export(train['path'].iloc[ind], format="wav")
        
    x = []
    for ind in tqdm(train.index):
        one_sec_silence = AudioSegment.silent(duration=1000)
        orig_audio = AudioSegment.from_file(train['path'].iloc[ind])
        if len(orig_audio) == 1000:
            pass
        else:
            audio = orig_audio.overlay(one_sec_silence)
            audio.export(train['path'].iloc[ind], format="wav")
        
        _x = get_spectrogram(train['path'].iloc[ind])
        x.append(_x)
    
    le = LabelEncoder()
    _y = le.fit_transform(train['labels'])
    y = to_categorical(_y, num_classes=train['labels'].unique().shape[0])
    
    with open('C:/Personal/Speech/label_encode.pkl', 'wb') as f:
        pickle.dump(le, f)
    
    return x, y

def get_model(input_shape):
    
    X_input = Input(shape = input_shape)
    X = Conv1D(filters=64, kernel_size=(5), strides=2)(X_input)
    X = BatchNormalization()(X)
    X = Activation(activation='relu')(X)
    X = Dropout(rate=0.2)(X)
    
    X = Conv1D(filters=128, kernel_size=(5), strides=2)(X)
    X = BatchNormalization()(X)
    X = Activation(activation='relu')(X)
    X = Dropout(rate=0.5)(X)
    
    X = Conv1D(filters=256, kernel_size=(5), strides=2)(X)
    X = BatchNormalization()(X)
    X = Activation(activation='relu')(X)
    X = Dropout(rate=0.5)(X)

    X = Flatten()(X)
    X = Dense(256, activation='relu')(X)
    X = Dense(128, activation='relu')(X)
    X = Dense(64, activation='relu')(X)
    X = Dense(32, activation='relu')(X)
    X = Dense(3, activation='softmax')(X)
    
    model = Model(inputs = X_input, outputs = X)
    return model
